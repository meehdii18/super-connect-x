Fonction checkValidPlay(entier : boardWidth, entier : boardHeight, tableau[boardWidth, boardHeight] grid, entier : playColumn) : booléen
Pré-conditions : 
	boardWidth > 0
	boardHeight > 0
	playColumn <= boardHeight
Post-conditions :
	valid = True si le coup est valide
	valid = False si le coup n'est pas valide
Variables :
	booléen : valid
Début :
	if(grid[boardHeight,playColumn] == 1 ou grid[boardHeight,playColumn] == 2)
		valid = False
	else 
		valid = True
	return valid
