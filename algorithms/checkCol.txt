Fonction checkCol(entier : requiredcoins, entier : boardWidth, entier : boardHeight, tableau[boardWidth, boardHeight] grid) : entier
Pré-conditions : 
	required coins > 0
	boardWidth > 0
	boardHeight > 0
Post-conditions :
	winner = 0 si pas de combinaison gagnante
	winner = 1 si le joueur gagne
	winner = 2 si l'ordinateur gagne
Variables :
entier row, col, count, winner;
Début :
	col <- 0
	TantQue ((col <= boardWidth)ET(count< requiredcoins)):
		current <- grid[0,col]
		row <- 0
		count <- 1
		TantQue ((row <= boardHeight)ET(count < requiredcoins)ET(boardHeight-row+count >= requiredcoins)ET(grid[row,col]!=0)):
			Si (current == grid[row,col]) alors :
				count <- count + 1
			Sinon:
				count <- 1
			FinSi
			current <- grid[row,col]
			row <- row + 1	
		FinTantQue
		col <- col + 1
	FinTantQue
	Si (count == requiredcoins) alors :
		winner <- current
	Sinon
		winner <- 0
	FinSi
	renvoyer winner
Fin
